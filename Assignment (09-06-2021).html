import matplotlib.pyplot as plt
import random
import time

def time_taken(n):
    start = time.time()
    lst=[]
    for x in range(0,n):
        n1 = random.randint(0,n)
        lst.append(n1)
    lst.sort()
    time.sleep(1)
    end = time.time()
    elapsed = end-start
    return elapsed
    
    


num = [50000,100000,150000,200000,250000]
times = []

for x in num:
    times.append(round(time_taken(x),4))

plt.title("Sorting List")
plt.xlabel("Number Of Elements")
plt.ylabel("Time Taken")
plt.plot(num,times)
plt.show()
